#here im making a function to find which store would give our model the best r^2 to work with.
#stores 36, 44 have some pretty high adjusted R^2

get_high_r2 = function(df) {
  number_of_stores = 45
  all_values = c()
  
  for (i in 1:number_of_stores) {
    single_store = df %>% filter(Store == i) %>% arrange(Date)
    mlr = lm(Weekly_Sales ~ factor(Holiday_Flag) + Temperature + Fuel_Price + CPI + Unemployment, data=single_store)
    all_values = c(all_values, summary(mlr)$adj.r.squared)
  }
  
  return(all_values)
}
